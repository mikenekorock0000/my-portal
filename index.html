// ========================================== // 【修正版】NISA投資信託処理（GASを経由せず直接取得してBotブロックを回避） // ========================================== async function fetchAndRenderNisa(widgetId, code, period) { const priceEl = document.getElementById('nisa-price-' + widgetId); if (!priceEl) return; if (!nisaDataCache[widgetId] || nisaDataCache[widgetId].code !== code) { priceEl.textContent = "取得中..."; try { // alloriginsという無料のプロキシを経由して、ブラウザから直接みんかぶのHTMLを取得 const targetUrl = 'https://itf.minkabu.jp/fund/' + encodeURIComponent(code); const proxyUrl = `https://api.allorigins.win/get?url=${encodeURIComponent(targetUrl)}`; const response = await fetch(proxyUrl); const data = await response.json(); if (!data.contents) throw new Error("取得失敗"); const html = data.contents; // JS側でHTMLからデータを抽出 const priceMatch = html.match(/class="[^"]*fw_b fz_l text_r[^"]*">([\d,]+).*?([+-]?[\d,]+)<\/span>円/i) || html.match(/([+-]?[\d,]+)円/); const diff = diffMatch ? parseInt(diffMatch[1].replace(/,/g, ''), 10) : 0; const history = []; const historyRegex = /\[(\d{13}),([\d\.]+)\]/g; let match; while ((match = historyRegex.exec(html)) !== null) { history.push({ date: parseInt(match[1], 10), price: parseFloat(match[2]) }); } history.sort((a, b) => a.date - b.date); if (currentPrice === 0) throw new Error("価格が見つかりません"); nisaDataCache[widgetId] = { code: code, data: { currentPrice, diff, history }, period: period }; } catch (e) { priceEl.textContent = "取得失敗"; document.getElementById('nisa-diff-' + widgetId).textContent = "Botブロック、またはURL不正"; return; } } renderNisaChartOnly(widgetId, period); } // ========================================== // 【修正版】RSS枠の処理（エラー時に真っ暗にならずメッセージを出すように強化） // ========================================== async function fetchRssWidget(id, urls) { const cont = document.getElementById('rss-'+id); if(!cont || !urls.length) return; try { const res = await Promise.all(urls.map(u => callGasApi({ action: 'rss', url: u.url }).then(r => ({ name: u.name, d: r.data })))); let items = []; res.forEach(r => { if(!r.d || r.d.includes('')) return; const xml = new DOMParser().parseFromString(r.d, 'text/xml'); const parsedItems = Array.from(xml.querySelectorAll('item')); parsedItems.forEach(i => items.push({ t: i.querySelector('title')?.textContent, l: i.querySelector('link')?.textContent, s: r.name, ts: new Date(i.querySelector('pubDate')?.textContent).getTime() })); }); // RSSの記事が見つからなかった場合のエラー表示 if (items.length === 0) { cont.innerHTML = '
記事が見つかりません。
登録したURLがXML形式(RSS)か確認してください。
'; return; } items.sort((a,b) => b.ts - a.ts); cont.innerHTML = '
    ' + items.slice(0, 6).map(i => `
  • ${i.s}${i.t}
    • `).join('') + '
'; } catch(e) { cont.innerHTML = '
通信エラーが発生しました。
'; } }
